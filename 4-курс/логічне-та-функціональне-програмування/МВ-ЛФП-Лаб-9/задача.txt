
; mapcar - застосовує функцію для конкретного елементу і повертає список з застосованою функцією
; довжина списку результату є така ж як вхідного списку

; конкретний приклад перетворює список градусів в список радіан
(print (mapcar (lambda (x) (* (/ x 180.0) pi)) '(90 180 270 360)))


; умова для фільтрації списку 
; як приклад видаляємо всі елементи які більше 5 
(defun less-than-5-p (x)
  (< x 5))

; mapcan - проходить по кожному елементу списку і зєднує список який повернула функція для елементу 
; з загальним списком результату. Тому довжина списока результату може != довжині вхідного списку 
( print (mapcan (lambda (x) (if (less-than-5-p x) (list x) '())) '(1 2 3 4 5 6)))
